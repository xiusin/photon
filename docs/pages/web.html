Controller 控制器
BaseController + 注解路由

import photon.web import veb // 定义 App 结构体（嵌入 BaseController + veb.Context） @[controller] pub struct App { web.BaseController veb.Context @[autowired] user_service &UserService } // GET 路由 @[get('/users')] pub fn (mut app App) list_users() veb.Result { users := app.user_service.find_all() return app.ok('{"data": ${users.json()}}') } // GET 带路径参数 @[get('/users/:id')] pub fn (mut app App) get_user(id string) veb.Result { user := app.user_service.find_by_id(id) or { return app.not_found('User not found') } return app.ok(user.json()) } // POST 路由 @[post('/users')] pub fn (mut app App) create_user() veb.Result { // 读取请求体 body := app.req.data.str() // ...解析并创建用户 return app.created('{"id": "new-id"}') } // PUT 路由 @[put('/users/:id')] pub fn (mut app App) update_user(id string) veb.Result { // ... return app.ok('{"updated": true}') } // DELETE 路由 @[delete('/users/:id')] pub fn (mut app App) delete_user(id string) veb.Result { // ... return app.no_content() } // Spring 等价：@RestController + @GetMapping / @PostMapping // Laravel 等价：Route::get('/users', [UserController::class, 'index'])

中间件链
MiddlewareChain · 可组合请求/响应处理
v
import photon.web // 1. 创建中间件链 mut chain := web.new_chain() // 2. 添加中间件（按顺序执行） chain.use(web.request_id_middleware()) // 注入 Request-ID chain.use(web.logging_middleware()) // 请求日志 chain.use(web.cors_middleware(cors_config)) // CORS chain.use(web.recovery_middleware()) // Panic 恢复 // 3. 自定义中间件 fn auth_middleware(ctx &web.MiddlewareContext) !bool { token := ctx.ctx.get_header('Authorization') or { '' } if token.len == 0 { return false // 中断链，返回 401 } // 验证 token... ctx.data['user_id'] = '42' // 传递数据给后续中间件/处理器 return true // 继续链 } chain.use(auth_middleware) // 4. 执行中间件链 mctx := web.new_middleware_context(veb_ctx) result := chain.execute(mctx)! // Spring 等价：HandlerInterceptor / Filter // Laravel 等价：Middleware $middleware

// 编程式路由注册 mut registry := web.new_route_registry() registry.register('GET', '/users', 'list_users') registry.register('POST', '/users', 'create_user') registry.register('GET', '/users/:id', 'get_user') registry.register('PUT', '/users/:id', 'update_user') registry.register('DELETE', '/users/:id', 'delete_user') // 便捷函数 route1 := web.get('/health', 'health_check') route2 := web.post('/login', 'login') route3 := web.put('/profile', 'update_profile') route4 := web.del('/cache', 'clear_cache') // RouterConfig mut config := web.RouterConfig{ base_path: '/api/v1' // 所有路由添加前缀 scan_package: 'myapp.controllers' enable_log: true }

CORS 跨域
细粒度跨域配置
v
mut cors_config := web.CorsConfig{ allowed_origins: ['https://example.com', 'https://app.example.com'] allowed_methods: ['GET', 'POST', 'PUT', 'DELETE'] allowed_headers: ['Content-Type', 'Authorization', 'X-Request-ID'] allow_credentials: true max_age: 3600 // preflight 缓存秒数 exposed_headers: ['X-Total-Count'] } chain.use(web.cors_middleware(cors_config))
🚀 生产实践
  • 使用 recovery_middleware 防止 panic 导致服务崩溃
  • request_id_middleware + Logger MDC 实现请求级日志追踪
  • CORS 配置不要使用 *，明确指定允许的 Origin
  • 中间件顺序很重要：recovery → logging → cors → auth → business
  • 路径参数使用 :id 风格，避免路由冲突
