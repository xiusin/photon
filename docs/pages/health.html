自定义指示器
实现 HealthIndicator 接口
v
// 实现 HealthIndicator 接口 pub struct RedisHealthIndicator { client &RedisClient } pub fn (h &RedisHealthIndicator) name() string { return 'redis' } pub fn (h &RedisHealthIndicator) check() health.Health { match h.client.ping() { ok { return health.new_health(.up).with_detail('version', h.client.info()) } err(e) { return health.new_health(.down).with_detail('error', e.str()) } } } // 注册 registry.register(&RedisHealthIndicator{client: redis})

// 配合 Web 模块暴露健康端点 @['/health'; get] pub fn (mut app App) health() veb.Result { agg := app.registry.check_all() if agg.status == .up { return app.json({'status': 'UP'}) } return app.status(503).json({'status': 'DOWN'}) } // K8s 探针配置： // livenessProbe: httpGet: path: /health port: 8080 // readinessProbe: httpGet: path: /health port: 8080
