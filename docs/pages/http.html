// 默认行为：4xx/5xx 抛 error resp := rt.get_for_entity('/not-found', {}) or { return error('请求失败: ${err}') } // 自定义错误策略（可插拔） rt2 := rt.set_error_handler(fn (resp http.ResponseEntity) ! { if resp.is_4xx() { return error('客户端错误 (${resp.status_code}): ${resp.body}') } if resp.is_5xx() { return error('服务端异常') } })

// 日志拦截器 logger := http.new_interceptor('logging', fn (e http.RequestEntity, next fn (http.RequestEntity) !http.ResponseEntity) !http.ResponseEntity { println('→ ${e.method} ${e.url}') resp := next(e) or { return err } println('← ${resp.status_code} (${resp.body.len} bytes)') return resp }) rt2 := rt.add_interceptor(logger) // 重试拦截器：对 5xx 自动重试（默认已带指数退避） retry_ic := http.new_interceptor('auto-retry', fn (e http.RequestEntity, next fn (http.RequestEntity) !http.ResponseEntity) !http.ResponseEntity { mut last_err := IError(none) for i in 0..3 { resp := next(e) or { last_err = err; continue } if resp.is_5xx() { continue } return resp } return last_err })

完整示例
用户 CRUD + 错误处理 + 拦截器

import photon.http // 配置客户端 rt := http.new_rest_template() .set_base_url('https://jsonplaceholder.typicode.com') .set_read_timeout(5000) .set_retry(3, 200) .add_interceptor(http.new_interceptor('log', unsafe { nil })) // GET — 获取用户列表 users := rt.get_for_object[[]User]('/users', {})! println(users.len) // GET — 获取单个用户（URI 模板） resp := rt.get_for_entity('/users/{id}', {'id': '1'})! user := resp.body_as[User]()! println(user.name) // POST — 创建用户（RequestEntity） entity := http.request_entity('POST', '/users') .body_json(CreateUser{name: 'Alice', email: 'alice@example.com'}) .header('Content-Type', 'application/json') resp := rt.exchange(entity)! created := resp.body_as[User]()! // PUT — 更新 rt.put('/users/{id}', '{"name":"Bob"}', {'id': '1'})! // DELETE rt.delete('/users/{id}', {'id': '1'})! // HEAD — 仅获取响应头 headers := rt.head_for_headers('/users', {})! println(headers['Content-Type'] or { '' })

设计要点
为什么这么设计
  • 不可变配置 — 所有 set_* 方法返回新实例，原始实例不变。可以安全地创建共享的 base 模板，再派生个性化副本。
  • 单一入口 — 所有 HTTP 方法（GET/POST/PUT/DELETE/PATCH/HEAD/OPTIONS）都委托给 exchange()，如同 Spring 的设计。
  • 零外部依赖 — 仅用 V 标准库 net.http、json、strings。无第三方 HTTP 库。
  • URI 模板 — 支持自定义分隔符，适用于非标准占位符格式。
  • 策略模式 — 错误处理和 URI 模板彻底可插拔，无需修改框架代码。
  • 职责链 — 拦截器按注册顺序递归包装，Spring 的 ClientHttpRequestInterceptor 风格。
