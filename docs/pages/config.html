import photon.config // 1. 创建配置实例 mut cfg := config.new() // 2. 添加配置源 cfg.add_source(config.new_file_source('config/app.json')) cfg.add_source(config.new_env_source()) // 环境变量 cfg.set_profile(['dev']) // 3. 加载并使用 cfg.load()! db_host := cfg.get_or('database.host', 'localhost') db_port := cfg.get_int_or('database.port', 5432) debug := cfg.get_bool_or('app.debug', true)

配置源
多源加载 · 按优先级合并

ConfigSource 是配置数据的来源接口，可组合多个源，后加载的覆盖先加载的同名键。

配置源说明Spring 等价Laravel 等价
FileSourceJSON 配置文件application.ymlconfig/*.php
EnvSource环境变量Environment variables.env
MapSource内存 Map（程序化配置）MapPropertySourceConfig::set()
FlagSource命令行参数CommandLinePropertySourceArtisan options

// 自定义 ConfigSource struct VaultSource { address string token string } pub fn (vs &VaultSource) load() !map[string]string { // 从 Vault API 拉取配置... return {'db.password': 'fetched-from-vault'} } pub fn (vs &VaultSource) name() string { return 'vault' } // 注册到配置 cfg.add_source(&VaultSource{address: 'https://vault.example.com', token: 's.xxx'})

Profile 环境
dev / staging / prod 环境隔离
v
mut cfg := config.new() // 设置激活的 Profile cfg.set_profile(['prod']) cfg.add_profile('us-east') // 可以叠加多个 // 加载时自动合并 Profile 特定的配置文件： // config/app.json → 基础配置 // config/app-prod.json → 生产环境覆盖 // config/app-us-east.json → 区域覆盖 cfg.load()! // Spring 等价：spring.profiles.active=prod,us-east // Laravel 等价：.env 中 APP_ENV=production

属性绑定
@[value] 注入 · 类型安全访问
v
@[component] pub struct DatabaseConfig { @[value('database.host')] host string @[value('database.port')] port int @[value('database.name')] name string @[value('database.pool.max')] max_pool int @[value('database.ssl.enabled')] ssl_enabled bool } // 也可直接从 Config 读取 host := cfg.get_or('database.host', 'localhost') // string port := cfg.get_int_or('database.port', 5432) // int ssl := cfg.get_bool_or('database.ssl', false) // bool rate := cfg.get_f64_or('rate.limit', 1.5) // f64 // Spring 等价：@Value("${database.host}") // Laravel 等价：config('database.host', 'localhost')

API 参考
Config 完整方法列表
方法返回值说明
config.new()&Config创建配置实例
cfg.add_source(source)—添加配置源
cfg.set_profile(profiles)—设置激活 Profile
cfg.add_profile(profile)—添加 Profile
cfg.load()!加载所有配置源并合并
cfg.get(key)string获取配置值（不存在返回空串）
cfg.get_or(key, default)string获取配置值或默认值
cfg.get_int(key)!int获取 int 值
cfg.get_int_or(key, default)int获取 int 值或默认值
cfg.get_bool(key)!bool获取 bool 值
cfg.get_bool_or(key, default)bool获取 bool 值或默认值
cfg.get_f64(key)!f64获取 f64 值
cfg.get_f64_or(key, default)f64获取 f64 值或默认值
🚀 生产实践
  • 敏感信息（密码、密钥）使用环境变量注入，不要写入配置文件
  • 环境变量覆盖文件配置（EnvSource 优先级最高）
  • 使用 @[value('key')] 绑定到 struct 字段，避免散落的魔术字符串
  • Profile 特定配置只覆盖差异项，减少重复配置
