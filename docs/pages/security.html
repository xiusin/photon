RBAC 权限
角色 · 权限 · 层级
v
// 定义角色和权限 mut rbac := security.new_rbac_manager() // 创建角色 rbac.add_role('admin') rbac.add_role('editor') rbac.add_role('viewer') // 分配权限 rbac.grant_permission('admin', 'user:delete') rbac.grant_permission('admin', 'user:create') rbac.grant_permission('editor', 'post:edit') rbac.grant_permission('viewer', 'post:read') // 角色继承（editor 继承 viewer 的权限） rbac.add_parent('editor', 'viewer') // 检查权限 has_perm := rbac.has_permission('editor', 'post:read') // true（继承自 viewer） // 使用注解保护端点 @[secured] // 需要认证 @[roles_allowed('admin')] // 需要 admin 角色 @[delete('/users/:id')] pub fn (mut app App) delete_user(id string) veb.Result { // ... } // Spring 等价：@Secured / @PreAuthorize("hasRole('admin')")

CSRF 防护
Double-Submit Cookie 模式
v
// 配置 CSRF csrf_config := security.CsrfConfig{ enabled: true cookie_name: 'XSRF-TOKEN' header_name: 'X-CSRF-TOKEN' form_field_name: '_csrf' token_length: 32 cookie_http_only: false // 前端 JS 需要读取 cookie_secure: true // HTTPS only cookie_same_site: 'Lax' ignored_methods: ['GET', 'HEAD', 'OPTIONS'] } // 使用 CookieCsrfTokenRepository mut repo := security.new_cookie_token_repo(csrf_config) // 生成 Token token := repo.generate_token() // 验证 Token（从请求 Header 或 Form 字段） valid := repo.validate_token(submitted_token) // Spring 等价：CsrfTokenRepository / CookieCsrfTokenRepository

SecurityFilterChain
与 Web 中间件链集成
v
// 构建安全过滤链 mut chain := security.new_security_filter_chain() // 公开端点 chain.permit('/api/auth/login', ['POST']) chain.permit('/api/auth/register', ['POST']) chain.permit('/api/health', ['GET']) // 需要认证的端点 chain.require_auth('/api/users/**') chain.require_role('/api/admin/**', 'admin') // 拒绝所有其他请求 chain.deny_all() // Spring 等价：SecurityFilterChain bean // Laravel 等价：Route::middleware('auth:api')
🚀 生产实践
  • JWT Secret 使用 256 位以上的随机密钥，从环境变量注入
  • Access Token 有效期设置 15-60 分钟，Refresh Token 7 天
  • CSRF 保护在生产环境必须开启，API 场景可关闭（使用 JWT 替代）
  • RBAC 角色设计遵循最小权限原则
  • 密码使用 bcrypt/argon2 哈希，永远不要明文存储
