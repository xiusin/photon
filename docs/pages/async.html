快速上手
创建执行器 · 提交任务 · 等待完成
v
import photon.async // 1. 创建执行器（4 个 worker 线程，队列容量 64） mut te := async.new_task_executor(4, 64) defer { te.shutdown() } // 2. 提交任务（阻塞式，队列满时产生背压） te.submit(fn () { send_email(user) })! // 3. 非阻塞提交（队列满时丢弃） te.try_submit(fn () { log_metrics() }) // 4. 等待所有任务完成 te.wait_all() // Spring 等价：@Async + TaskExecutor // Laravel 等价：dispatch(new ProcessPodcast($podcast))

@[async] 注解
编译期扫描 · 零运行时反射
v
@[service] pub struct EmailService { @[autowired] executor &async.TaskExecutor } @[async] pub fn (s &EmailService) send_welcome(user User) { // 该方法将异步执行 s.mailer.send(user.email, 'Welcome', '...')! } // 编译期发现所有 @[async] 方法 methods := async.extract_async_methods[EmailService]()

优雅关闭
排空任务 · 等待完成
v
// shutdown() 会： // 1. 拒绝新任务提交 // 2. 等待已提交任务完成 // 3. 关闭所有 worker 线程 te.shutdown() // 也可手动分步： te.stop_accepting() // 停止接受新任务 te.wait_all() // 等待剩余任务完成
