Retryable · 执行策略 · 退避
v
import photon.retry // 1. 基础重试 — 最多重试 3 次 result := retry.run(max_attempts: 3, fn () string { return risky_operation() }) or { // 全部失败后返回 error println('所有重试均失败: ${err}') } // 2. 固定退避 — 每次等待 100ms result2 := retry.run( max_attempts: 5 delay: 100 * time.millisecond fn () int { return call_remote_api() } ) or { 0 } // 3. 指数退避 + 抖动 result3 := retry.run( max_attempts: 5 initial_delay: 100 * time.millisecond multiplier: 2.0 // 100 → 200 → 400 → 800 jitter: 0.5 // ±50% 随机抖动 fn () string { return call_flaky_service() } ) or { '' }

@[retryable] 注解
编译期扫描 · 声明式重试
v
// 编译期通过 $for 扫描 @[retryable] 注解 @[retryable: 3; delay=100ms] pub fn call_payment_gateway(order string) PaymentResult { // 首次失败将自动重试，最多 3 次 return http_client.post('/pay', order) or { return PaymentResult{ 'failed' } } } // 指数退避注解 @[retryable: 5; exponential; initial_delay=200ms; multiplier=2.0; jitter=0.3] pub fn sync_inventory() SyncResult { return inventory_api.fetch() }
