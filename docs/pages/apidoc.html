import photon.apidoc // 1. 初始化 — 返回 Store + Collector store, coll := apidoc.init('./data/apidoc') or { panic(err) } // 2. Collector — 收集端点信息 coll.add_endpoint( method: 'GET' path: '/api/users/:id' summary: '获取用户详情' tags: ['user'] response_type: 'User' ) // 3. Store — 持久化与查询 endpoints := store.list_endpoints() println(endpoints.len) // 已收集的端点数量 // 4. 仅初始化 Store（无 Collector） store2 := apidoc.init_store('./data/apidoc') or { panic(err) }

标准 JSON 响应
encode_response · api_error
v
// 成功响应 — {"code":0,"msg":"ok","data":{...}} json := apidoc.encode_response(0, 'ok', '{"id":1,"name":"alice"}') println(json) // {"code":0,"msg":"ok","data":{"id":1,"name":"alice"}} // 错误响应 — {"code":404,"msg":"not found"} err_json := apidoc.api_error(404, 'not found') println(err_json) // {"code":404,"msg":"not found"}

// 在 Controller 中使用 Collector 自动收集 @['/api/users/:id'; get] pub fn (mut app App) get_user(id int) veb.Result { user := user_repo.find_by_id(id) or { return app.json(apidoc.api_error(404, err.msg())) } return app.json(apidoc.encode_response(0, 'ok', user.json_str())) } // 暴露文档端点 @['/apidoc'; get] pub fn (mut app App) api_docs() veb.Result { return app.json(app.store.list_endpoints()) }
